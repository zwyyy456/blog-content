## Description
[2331.evaluate-boolean-binary-tree](https://leetcode.com/problems/evaluate-boolean-binary-tree/)

## Solution
dfs(recursion)

## Code
```cpp
class Solution {
public:
    bool dfs(TreeNode *root) {
        if (root->left == nullptr) {
            return root->val;
        }
        if (root-> val == 2) {
            return dfs(root->left) || dfs(root->right);
        } else {
            return dfs(root->left) && dfs(root->right);
        }
    }
    bool evaluateTree(TreeNode* root) {
        return dfs(root);
    }
};
```
